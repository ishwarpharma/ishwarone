function highlightText(text, tokens) { let result = String(text); tokens.forEach(token => { if (!token) return; const esc = token.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'); result = result.replace(new RegExp('(' + esc + ')', 'gi'), '$1'); }); return result; } function scoreToken(brand, company, token) { let score = 0, matched = false; if (brand.startsWith(token)) { score += 10; matched = true; } else { const bw = brand.split(/[\s\-\/\(]+/); if (bw.some(w => w && w.startsWith(token))) { score += 5; matched = true; } else if (brand.includes(token)) { score += 1; matched = true; } } if (company.startsWith(token)) { score += 4; matched = true; } else { const cw = company.split(/[\s\-\/\(]+/); if (cw.some(w => w && w.startsWith(token))) { score += 2; matched = true; } else if (company.includes(token)) { score += 0.5; matched = true; } } return [score, matched]; } function filterProducts() { const input = document.getElementById("search").value.trim(); const container = document.getElementById("suggestions"); container.innerHTML = ""; if (!input) return; const tokens = input.toLowerCase().split(/\s+/).filter(Boolean); const scored = stockData.map(p => { const brand = (p.brand || "").toLowerCase(); const company = (p.company || "").toLowerCase(); let totalScore = 0, allMatched = true; for (const token of tokens) { const [s, matched] = scoreToken(brand, company, token); totalScore += s; if (!matched) allMatched = false; } if (allMatched) totalScore += 50; return { p, score: totalScore }; }) .filter(x => x.score > 0) .sort((a, b) => b.score - a.score); scored.slice(0, 200).forEach(({ p }) => { const div = document.createElement("div"); div.className = "product-card"; const hlBrand = highlightText(p.brand || "", tokens); const hlCompany = highlightText(p.company || "", tokens); div.innerHTML = ` ${hlBrand} | ${hlCompany} | Rate: ${Number(p.rate).toFixed(2)} | MRP: ${Number(p.mrp).toFixed(2)} | Stock: ${p.stock || "-"} `; div.onclick = () => { selectedProduct = p; document.getElementById("entryForm").style.display = "block"; document.getElementById("rateInfo").innerHTML = `\u{1F4B0} Rate: \u20B9${p.rate}    \u{1F3F7}\uFE0F MRP: \u20B9${p.mrp}`; container.innerHTML = ""; document.getElementById("search").value = `${p.brand} | ${p.company}`; }; container.appendChild(div); }); }
